\section*{Basic Definitions}
\subsection*{Binary Operation}

\subsection*{Groups}
\textbf{Definition}: A group $(G,\cdot)$ is a set $G$ and binary operation $\cdot$ satisfying:
\begin{itemize}
	\item Associativity: $\forall_{a,b,c\in G}(a*b)*c=a*(b*c)$ 
	\item Identity element: $\exists e\in G$ satisfying $e\cdot a = a = a \cdot e$.
	\item Inverse: $\forall_{a\in G}\exists_{b\in G} ab =ba = e$
\end{itemize}

\subsection*{Commutative}

\subsection*{Associative}

A group $(G,*)$ is associative if, for any $\forall u,v,w\in G$: 
\[
(u*v)*w = u*(v*w)	
\]

\subsection*{Isomorphism}
Let $(G,*)$ and $(H,*)$ be groups, if there exists a $f:G\to H$ which is bijective. Then $G$ and $H$ are isomorphic.

\section*{Basic Properties}
\subsection*{Cancelation}
Theorem 1.3:  Let $(G,*)$ be a group, then: 
\begin{enumerate}
	\item The identity element $e$ exists and is unique. 
	\item Any element $a$ has a unique inverse $b$ where  $ab=ba=e$.
\end{enumerate}
